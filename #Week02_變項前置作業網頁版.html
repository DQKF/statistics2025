說明：在統計分析中，如果某一分類的樣本數太小（例如 n < 30，甚至 < 5），會造成：
1.估計不穩定:回歸或平均比較中，該類別的估計值會有極大標準誤。
2.假設檢定失真:卡方檢定 (Chi-square test) 要求各格期望值至少 >5；

5) 建立新的年齡 column "age_N"
C30 <- C30 %>%
  mutate(                            # mutate() 新增或變更資料欄位的函數。
    age = as.numeric(age),           # 求慎重，再處理一次age變項確立為numeric。
    age_N = case_when(
      age >= 18 & age <= 29 ~ "20s", # 為求便利，將18,19歲改列為 20s。
      age >= 30 & age <= 39 ~ "30s", # 因 age 為 numeric，非character & factor，
      age >= 40 & age <= 49 ~ "40s", # 可不加括號。但新成立的變項age_N，為factor
      age >= 50 & age <= 59 ~ "50s", # 變項，引此需要加括號 " "。
      age >= 60             ~ "60+",
      TRUE                  ~ NA_character_
    ),
    age_N = factor(age_N, # 將 新的column age_N 設定為 factor。
    levels = c("20s","30s","40s","50s","60+"), # levels()，從低到高，設定ordinal
    ordered = TRUE) # 將age_N 設置為順序尺度 (Ordinal Scale)
       )
